  ```

    # 생성
    self.rd = redis.StrictRedis(host='localhost', port=F.config['redis_port'], db=self.db_item.id)

    # DB 초기화
    self.rd.flushdb()

    # set
    self.rd.set(key, json.dumps(data))

    # get
    self.rd.get(key)
  ```
